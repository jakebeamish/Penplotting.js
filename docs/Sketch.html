Sketch

Source
  • Sketch.js, line 107
Example
import { Sketch, Line, Circle } from "@jakebeamish/penplotting";

const sketch = new Sketch({
    backgroundColor: "#ffffff"
});

sketch.generate = () => {
    const line = Line.fromArray([0, 0, 100, 100]);
    const circlesArray = [
        new Circle(10, 10, 10),
        new Circle(40, 40, 15),
        new Circle(80, 80, 20)
    ];
    sketch.add([line, circlesArray]);
}

sketch.draw();
